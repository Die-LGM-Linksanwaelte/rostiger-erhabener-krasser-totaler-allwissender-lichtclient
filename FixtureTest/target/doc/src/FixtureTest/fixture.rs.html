11/// Represents the various configurable properties of a lighting fixture.
12///
13/// This enum encapsulates all supported attribute types of a fixture,
14/// allowing each property to carry its associated DMX channel(s).
15/// It provides a unified way to describe colors, movement, beam effects,
16/// gobos, atmospheric controls, and any custom or manufacturer-specific
17/// attributes.
18///
19/// # Variants
20///
21/// * **Color(Color)**
22///   A color-related property such as RGB, CMY, or other color-mixing systems.
23///
24/// * **Dimmer(Channel)**
25///   Controls fixture brightness (0–255).
26///
27/// * **Strobe(Channel)**
28///   Controls strobe rate or shutter pulse effects.
29///
30/// * **Beam { zoom, focus, frost }**
31///   Beam-shaping properties:
32///   - `zoom`: controls beam width
33///   - `focus`: controls sharpness
34///   - `frost`: applies diffusion/frost effect
35///
36/// * **Shutter(Channel)**
37///   Mechanical shutter control (open/close).
38///
39/// * **Prism { prism, prism_rotation, prism_indexation }**
40///   Prism and prism-effect controls:
41///   - `prism`: enables/selects prism
42///   - `prism_rotation`: rotation speed/direction
43///   - `prism_indexation`: discrete index positioning
44///
45/// * **Gobo { gobo_rotation, gobo_rotation_speed, gobo_wheel_rotation, gobo_wheel_rotation_speed }**
46///   Gobo selection and motion:
47///   - `gobo_rotation`: absolute rotation
48///   - `gobo_rotation_speed`: continuous rotation speed
49///   - `gobo_wheel_rotation`: selects wheel slot rotation
50///   - `gobo_wheel_rotation_speed`: rotation speed of the gobo wheel
51///
52/// * **Position { pan, tilt }**
53///   Movement parameters for head-positioning.´
54///
55/// * **UV(Channel)**
56///   UV-LED intensity control.
57///
58/// * **Speed(Channel)**
59///   Global macro speed or effect speed.
60///
61/// * **Fog { fog_intensity, fog_fan_speed }**
62///   Atmospheric effects:
63///   - `fog_intensity`: fog output amount
64///   - `fog_fan_speed`: fan speed for fog dispersion
65///
66/// * **Other(String, Channel)**
67///   Any manufacturer-specific or unsupported property, given as a descriptive name and channel index.
68enum Property {
69    Color(Color),
70    Dimmer(Channel),
71    Strobe(Channel),
72    Beam{
73        zoom: Channel,
74        focus: Channel,
75        frost: Channel,

77    Shutter(Channel),
78    Prism{
79        prism: Channel,
80        prism_rotation: Channel,
81        prism_indexation: Channel,
82    },
83    Gobo{
84        gobo_rotation: Channel,
85        gobo_rotation_speed: Channel,
86        gobo_wheel_rotation: Channel,
87        gobo_wheel_rotation_speed: Channel,
88    },
89    Position{
90        pan: Channel,
91        tilt: Channel,
92    },
93    UV(Channel),
94    Speed(Channel),
95    Fog{
96        fog_intensity: Channel,
97        fog_fan_speed: Channel,
98    },
99
100    Other(String, Channel),
101}
